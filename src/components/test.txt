import React from 'react'
import { useState, useEffect } from 'react';
import { Container, Row, Col } from 'react-bootstrap';
import Carousel from "react-multi-carousel";
import "react-multi-carousel/lib/styles.css";
import skill1 from '../assets/html.png'
import './style.css'
const Skill = () => {
    const [activeItem, setActiveItem] = useState(0)
    const responsive = {
        superLargeDesktop: {
            breakpoint: { max: 4000, min: 3000 },
            items: 5
        },
        desktop: {
            breakpoint: { max: 3000, min: 1024 },
            items: 3
        },
        tablet: {
            breakpoint: { max: 1024, min: 464 },
            items: 2
        },
        mobile: {
            breakpoint: { max: 464, min: 0 },
            items: 1
        }
    };
    const handleSlideChange = (currentIndex) => {
        setActiveItem(currentIndex);
    };
    useEffect(() => {
        const interval = setInterval(() => {
            const nextItem = (activeItem + 1) % 4;
            setActiveItem(nextItem);
        }, 5000);
        return () => clearInterval(interval);
    }, [actvieItem]);
    return (

        <section className='skill skill-bg' id='skills'>
            <Container>
                <Row>
                    <Col>
                        <div className='skill-box'>
                            <h2>Skills</h2>
                            <p>Dream skills: Full-stack, frameworks, UX, performance, tech fusion in programming and web.</p>
                            <Carousel responsive={responsive}
                                infinite={true}
                                className='skill-slider'
                                customTransition='transform 500ms ease-in-out'
                                afterChange={handleSlideChange}
                                containerClass='carousel-container'
                                itemClass='carousel-item'
                                keyBoardControl={false}
                                transitionDuration={500}
                                arrows={true} // Keep the arrows for manual navigation
                                showDots={false}
                                sliderClass=''
                                draggable={true} // Enable manual dragging
                                swipeable={true} // Enable swipe on mobile
                                autoPlaySpeed={5000}
                                centerMode={false}
                                removeArrowOnDeviceType={[]} // Remove arrows only for desktop view
                                dotListClass=''>
                                <div className='item'>
                                    <img src={skill1} alt="bootstrap" />
                                    <h5>Web development</h5>
                                </div>
                                <div className='item'>
                                    <img src={skill1} alt="html" />
                                    <h5>Web App development</h5>
                                </div>
                                <div className='item'>
                                    <img src={skill1} alt="CSS" />
                                    <h5>Front End development</h5>
                                </div>
                                <div className='item'>
                                    <img src={skill1} alt="React" />
                                    <h5>skill development</h5>
                                </div>

                            </Carousel>
                        </div>
                    </Col>
                </Row>
            </Container>
        </section>
    )
}

export default Skill
